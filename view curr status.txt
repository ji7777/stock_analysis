
  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##TRADER"."VW_CHECK_CURRSTATUS" ("STOCK_NAME", "TOTAL_QTY", "BUY_AVG", "TOTAL_BUY_AMOUNT", "BUY_CHARGE", "CURR_VALUE", "CAL_CURRVALUE", "CAL_PROFIT", "CALC_PERC") AS 
  select STOCK_NAME, TOTAL_QTY,BUY_AVG, TOTAL_BUY_AMOUNT, BUY_CHARGE, CURR_VALUE, CAL_CURRVALUE, CAL_PROFIT, CALC_PERC
from pl_stock where time_key=to_number(to_char(SYSDATE,'YYYYMMDD')) and status<>'SOLD';
